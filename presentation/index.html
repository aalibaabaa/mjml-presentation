Mail Jet Markup Language - MJML

The idea

  • Developed and maintained by MailJet
  • Create email templates quick and easy
  • Make building responsive emails easy
  • Do it component based

MJML structure

            
              <mjml>
                <mj-head>
                  ...
                </mj-head>
                <mj-body>
                  ...
                </mj-body>
              </mjml>
  • mjml
  • head (global settings)
  • body (everything you see)

                <mj-body>

Tutorial:
tutorial-creating-your-own-mj-component

Boilerplate:
https://github.com/mjmlio/mj-component-boilerplate 

          
            import { MJMLElement } from 'mj-core'
            import React, { Component } from 'react'
            import reverse from 'lodash/reverse'
            import Section from 'mj-section'
            import Column from 'mj-column'
            import Image from 'mj-image'
            import Text from 'mj-text'

            // Tag Name
            const tagName = 'mj-column-image-text'
            // List of parent tags
            const parentTag = ['mj-container']
            // If false, the component can contain MJML; if true, it can contain only plain HTML
            const endingTag = false
            const defaultMJMLDefinition = {
              content: '',
              attributes: {
                'direction': 'left',
                'color': 'blue',
                'font-size': '10px',
                'image-padding': 0,
                'image-src': null,
                'image-width': null,
                'column-width': '50%',
                'text': '',
              },
            }

            @MJMLElement //Decorator
            class ColumnImageText extends Component {

              renderImage () {
                const { mjAttribute } = this.props

                return (
                  <Column key="image" width={mjAttribute('column-width')}>
                    <Image
                      padding={mjAttribute('image-padding')}
                      width={mjAttribute('image-width')}
                      src={mjAttribute('image-src')}
                    />
                  </Column>
                )
              }

              renderContent () {
                const { mjAttribute } = this.props

                return (
                  <Column key="content" width={mjAttribute('column-width')}>
                    <Text font-size={mjAttribute('font-size')} color={mjAttribute('color')}>
                      { mjAttribute('text') }
                    </Text>
                  </Column>
                )
              }

              render () {
                const { mjAttribute } = this.props
                const columns = [this.renderImage(), this.renderContent()]

                return (
                  <Section {...this.props}>
                    { mjAttribute('direction') == 'right' ? reverse(columns) : columns }
                  </Section>
                )

              }

            }

            ColumnImageText.tagName = tagName
            ColumnImageText.parentTag = parentTag
            ColumnImageText.endingTag = endingTag
            ColumnImageText.defaultMJMLDefinition = defaultMJMLDefinition

            export default ColumnImageText
